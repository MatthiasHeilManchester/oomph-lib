list(APPEND CMAKE_MESSAGE_INDENT " ")
message(VERBOSE "Entered matrix_concatenation subdirectory")

cmake_minimum_required(VERSION 3.22 FATAL_ERROR)
project(matrix_concatenation C CXX Fortran)
if(NOT oomphlib_FOUND)
  find_package(oomphlib CONFIG REQUIRED PATHS "../../../install")
endif()
include(CTest)

oomph_add_executable(
  NAME matrix_concatenation
  SOURCES matrix_concatenation.cc
  LIBRARIES oomph::meshes oomph::generic)

# ---------------------------------- TESTING ----------------------------------

include(ProcessorCount)
ProcessorCount(NUM_PROCESSORS)
if(NUM_PROCESSORS GREATER_EQUAL 4)
  oomph_add_test(
    TEST_NAME self_test.mpi.matrix_concatenation
    DEPENDS_ON matrix_concatenation
    COMMAND ./validate.sh ${OOMPH_ROOT_DIR} ${OOMPH_MPI_VARIABLENP_RUN_COMMAND}
    TEST_FILES validate.sh validata
    LABELS mpi matrix)

  # Don't run this test alongside any other tests
  set_tests_properties(self_test.mpi.matrix_concatenation PROPERTIES PROCESSORS
                                                                     4)
else()
  message(
    STATUS
      "Skipping test 'self_test.mpi.matrix_concatenation' as you don't have enough processors available."
  )
endif()

message(VERBOSE "Leaving matrix_concatenation subdirectory")
