cmake_minimum_required(VERSION 3.22 FATAL_ERROR)
project(vector_split_without_communication C CXX Fortran)
if(NOT oomphlib_FOUND)
  find_package(oomphlib CONFIG REQUIRED PATHS "../../../install")
endif()
include(CTest)

oomph_add_executable(
  NAME vector_split_without_communication
  SOURCES vector_split_without_communication.cc
  LIBRARIES oomph::meshes oomph::generic)

# ---------------------------------- TESTING ----------------------------------

include(ProcessorCount)
ProcessorCount(NUM_PROCESSORS)
if(NUM_PROCESSORS GREATER_EQUAL 4)
  oomph_add_test(
    TEST_NAME self_test.mpi.vector_split_without_communication
    DEPENDS_ON vector_split_without_communication
    COMMAND ./validate.sh ${OOMPH_ROOT_DIR} ${OOMPH_MPI_VARIABLENP_RUN_COMMAND}
    TEST_FILES validate.sh validata
    LABELS mpi vector)

  # Don't run this test alongside any other tests
  set_tests_properties(self_test.mpi.vector_split_without_communication
                       PROPERTIES PROCESSORS 4)
else()
  message(
    STATUS
      "Skipping test 'self_test.mpi.vector_split_without_communication' as you don't have enough processors available."
  )
endif()
