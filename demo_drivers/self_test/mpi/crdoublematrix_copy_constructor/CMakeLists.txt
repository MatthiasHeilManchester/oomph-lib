    LABELS mpi matrix)

  set_tests_properties(self_test.mpi.crdoublematrix_copy_constructor
                       PROPERTIES PROCESSORS 4)

    STATUS
      "Skipping test 'self_test.mpi.crdoublematrix_copy_constructor' as you don't have enough processors available."
  )
