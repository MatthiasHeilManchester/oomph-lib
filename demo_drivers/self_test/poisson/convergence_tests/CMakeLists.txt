  NAME q_convergence_2d

  NAME t_convergence_2d
  SOURCES t_convergence_2d.cc
  LIBRARIES oomph::poisson oomph::meshes oomph::generic)

  NAME t_convergence_3d
  SOURCES t_convergence_3d.cc
  LIBRARIES oomph::poisson oomph::meshes oomph::generic)
